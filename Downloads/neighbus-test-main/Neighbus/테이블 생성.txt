-- 데이터베이스 생성
CREATE DATABASE `neighbus` DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

USE `neighbus`;

-- 유저 등급 테이블 (grades)
CREATE TABLE `grades`(
    `id` INT PRIMARY KEY COMMENT '등급 ID (0: 관리자, 1: 비인증, 2: 인증, 3: 신뢰)',
    `grade` VARCHAR(10) NOT NULL COMMENT '등급명 (예: 관리자, 비인증, 인증, 신뢰)'
);
 
-- 도, 특별시 테이블 (provinces)
CREATE TABLE `provinces`(
    `id` INT PRIMARY KEY COMMENT '시/도 ID',
    `province` VARCHAR(50) COMMENT '시/도 이름 (예: 서울특별시, 경기도)'
);

-- 시, 군, 구 테이블 (cities)
CREATE TABLE `cities`(
    `id` INT PRIMARY KEY COMMENT '시/군/구 ID',
    `province` INT NOT NULL COMMENT 'FK: 소속된 provinces.id (시/도 ID)',
    `city` VARCHAR(50) NOT NULL COMMENT '시/군/구 이름 (예: 강남구, 수원시)'
);

-- 유저 테이블 (users)
CREATE TABLE `users`(
    `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '유저 고유 ID',
    `name` VARCHAR(20) NOT NULL COMMENT '이름',
    `username` VARCHAR(500) NOT NULL UNIQUE COMMENT '로그인 아이디',
    `password` VARCHAR(100) NOT NULL COMMENT '비밀번호',
    `province` INT NOT NULL COMMENT 'FK: 시/도 테이블 참조 id',
    `city` INT NOT NULL COMMENT 'FK: 거주 지역 cities.id',
    `address` VARCHAR(200) NOT NULL COMMENT '상세 주소',
    `phone` VARCHAR(11) NOT NULL COMMENT '전화번호 (하이픈 제외 11자리)',
    `email` VARCHAR(50) NOT NULL UNIQUE COMMENT '이메일 주소',
    `image` VARCHAR(100) DEFAULT '/img/profile/default-profile.png' COMMENT '프로필 이미지 파일 경로/이름',
    `grade` INT DEFAULT 1 NOT NULL COMMENT 'FK: 유저 등급 grades.id (기본값 1: 비인증)',
    `birth` VARCHAR(6) NOT NULL COMMENT '생년월일 (YYMMDD 형식)',
    `sex` VARCHAR(1) NOT NULL COMMENT '성별 (M: 남자, F: 여자)',
    `user_uuid` VARCHAR(36) NOT NULL COMMENT '친구 추가를 위한 고유 UUID',
    `nickname` VARCHAR(20) NULL COMMENT '닉네임',
    `created_at` DATETIME DEFAULT NOW() COMMENT '가입일시',
    `provider` VARCHAR(20) NULL ,
	`provider_id` VARCHAR(100) NULL ,
	`is_blocked` TINYINT(1) DEFAULT 0 COMMENT '0:정상 1:정지',
	`blocked_until` DATETIME NULL COMMENT '정지 종료일',
    FOREIGN KEY (`province`) REFERENCES `provinces`(`id`),
    FOREIGN KEY (`city`) REFERENCES `cities`(`id`),
    FOREIGN KEY (`grade`) REFERENCES `grades`(`id`)
);
-- 알림 테이블
CREATE TABLE notification (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    users_id INT NOT NULL,                 
    notification_type VARCHAR(50) NOT NULL,
    content VARCHAR(255) NOT NULL,
    url VARCHAR(255),
    is_read TINYINT(1) DEFAULT 0,
    created_at DATETIME DEFAULT NOW(),
    
    CONSTRAINT fk_notification_users 
    FOREIGN KEY (users_id) REFERENCES users(id) ON DELETE CASCADE 
);

-- 신고 테이블
CREATE TABLE reports (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '신고 고유 ID',
    type VARCHAR(20) NOT NULL COMMENT '신고 유형 (예: CLUB, GALLERY, GATHERING, POST, COMMENT)',
    reporter_id INT NOT NULL COMMENT '신고자 ID',
    target_id BIGINT NOT NULL COMMENT '신고 대상 ID (게시글ID, 댓글ID 등)',
    reason VARCHAR(255) NOT NULL COMMENT '신고 사유',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '신고일',
    status VARCHAR(20) DEFAULT 'PENDING' COMMENT '처리 상태 (PENDING, PROCESSING, COMPLETED)',
    FOREIGN KEY (reporter_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 1. 친구 테이블 (friends) - 수정됨
-- 유저(users)가 삭제되면, 친구 관계 데이터도 자동으로 삭제되도록 CASCADE 추가
CREATE TABLE `friends`(
    `my_id` INT NOT NULL COMMENT 'FK: 내 아이디. 유저ID 참조 (users.id)',
    `friend_id` INT NOT NULL COMMENT 'FK: 친구 아이디. 유저 ID 참조 (users.id)',
    FOREIGN KEY (`my_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,       -- ★ 수정됨
    FOREIGN KEY (`friend_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,   -- ★ 수정됨
    PRIMARY KEY(`my_id`, `friend_id`)
);

-- 2. 친구 요청 테이블 (friend_state) - 수정됨
-- 유저(users)가 삭제되면, 요청 데이터도 자동으로 삭제되도록 CASCADE 유지
CREATE TABLE `friend_state`(
    `sender` INT NOT NULL COMMENT 'FK: 친구 요청을 보낸 유저 ID (users.id)',
    `receiver` INT NOT NULL COMMENT 'FK: 친구 요청을 받은 유저 ID (users.id)',
    `state` INT DEFAULT 1 COMMENT '친구 상태 (1: 요청 중, 2: 수락됨/친구, 3: 거절)',
    `created_at` datetime DEFAULT now() COMMENT '요청일',
    FOREIGN KEY (`sender`) REFERENCES `users`(`id`) ON DELETE CASCADE,      -- ★ 유지/수정
    FOREIGN KEY (`receiver`) REFERENCES `users`(`id`) ON DELETE CASCADE,    -- ★ 유지/수정
    PRIMARY KEY(`sender`, `receiver`)
);

-- 카테고리 테이블 (club_categorys)
CREATE TABLE `club_categorys`(
    `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '카테고리 고유 ID',
    `name` VARCHAR(20) NOT NULL COMMENT '카테고리 종류'
);

-- 동아리 테이블 (clubs)
CREATE TABLE `clubs`(
    `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '동아리 고유 ID',
    `club_name` VARCHAR(100) NOT NULL UNIQUE COMMENT '동아리 이름',
    `write_id` int NOT NULL COMMENT '동아리 생성자',
    `club_info` VARCHAR(200) NOT NULL COMMENT '동아리 상세정보',
    `category` INT NOT NULL COMMENT '카테고리',
    `province_id` INT NOT NULL COMMENT '동아리 지역',
    `city` INT NOT NULL COMMENT 'FK: 동아리 활동 동네 cities.id',
    `club_img` VARCHAR(100) NOT NULL COMMENT '동아리 이미지',
    `created_at` DATETIME DEFAULT NOW() COMMENT '생성일시',
    FOREIGN KEY (`province_id`) REFERENCES `provinces`(`id`),
    FOREIGN KEY (`city`) REFERENCES `cities`(`id`),
    FOREIGN KEY (`write_id`) REFERENCES `users`(`id`),
    FOREIGN KEY (`category`) REFERENCES `club_categorys`(`id`)
);

-- 동아리 멤버 테이블 (club_members)
CREATE TABLE `club_members`(
    `club_id` INT NOT NULL COMMENT 'FK: 동아리 ID (clubs.id)',
    `user_id` INT NOT NULL COMMENT 'FK: 멤버 유저 ID (users.id)',
    `created_at` DATETIME DEFAULT NOW() COMMENT '동아리 가입일시',
    FOREIGN KEY (`club_id`) REFERENCES `clubs`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,
    PRIMARY KEY(`club_id`, `user_id`)
);


-- 자유게시판 테이블 (freeboards)
CREATE TABLE `freeboards`(
    `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '게시글 고유 ID',
    `title` VARCHAR(100) NOT NULL COMMENT '게시글 제목',
    `content` LONGTEXT NOT NULL COMMENT '게시글 내용',
    `writer` INT NOT NULL COMMENT 'FK: 작성자 유저 ID (users.id)',
    `club_id` INT NOT NULL COMMENT 'FK: 동아리 고유 ID',
    `view_count` INT DEFAULT 0 COMMENT '조회수',
    `created_at` DATETIME DEFAULT NOW() COMMENT '작성일시',
    FOREIGN KEY (`writer`) REFERENCES `users`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`club_id`) REFERENCES `clubs`(`id`) ON DELETE CASCADE
);

-- 자유게시판 이미지 테이블 (freeboard_images)
CREATE TABLE `freeboard_images`(
    `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '이미지 고유 ID',
    `freeboard` INT NOT NULL COMMENT 'FK: 소속된 freeboards.id',
    `img` VARCHAR(50) NOT NULL COMMENT '이미지 파일 이름 (UUID.확장자)',
    FOREIGN KEY (`freeboard`) REFERENCES `freeboards`(`id`) ON DELETE CASCADE
);

-- 자유게시판 댓글 테이블 (freeboard_comments)
CREATE TABLE `freeboard_comments`(
    `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '댓글 고유 ID',
    `freeboard` INT NOT NULL COMMENT 'FK: 소속된 freeboards.id',
    `parent` INT DEFAULT 0 NOT NULL COMMENT 'FK: 부모 댓글 ID (대댓글 시 참조)',
    `writer` INT NOT NULL COMMENT 'FK: 작성자 유저 ID (users.id)',
    `content` VARCHAR(500) NOT NULL COMMENT '댓글 내용',
    `created_at` DATETIME DEFAULT NOW() COMMENT '작성일시',
    FOREIGN KEY (`freeboard`) REFERENCES `freeboards`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`writer`) REFERENCES `users`(`id`) ON DELETE CASCADE
);

-- 자유게시판 좋아요/싫어요 테이블 (freeboard_reactions)
CREATE TABLE `freeboard_reactions` (
    `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '반응 고유 ID',
    `freeboard_id` INT NOT NULL COMMENT 'FK: 게시글 ID (freeboards.id)',
    `user_id` INT NOT NULL COMMENT 'FK: 유저 ID (users.id)',
    `reaction_type` int NOT NULL COMMENT '1:좋아요  2:싫어요',
    `created_at` DATETIME DEFAULT NOW() COMMENT '반응 등록일시',
    FOREIGN KEY (`freeboard_id`) REFERENCES `freeboards`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,
    UNIQUE KEY `unique_user_reaction` (`freeboard_id`, `user_id`) COMMENT '한 유저는 게시글당 하나의 반응만 가능'
) COMMENT='자유게시판 좋아요/싫어요 반응 테이블';

-- 모집 테이블 (recruitments)

CREATE TABLE chat_room (
    room_id VARCHAR(50) NOT NULL COMMENT '채팅방 식별자 (모집: "15", 친구: "3_5")',
    room_name VARCHAR(100) NOT NULL COMMENT '채팅방 이름',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성 시간',
    
    -- [추가] 자동 삭제를 위한 연결 고리들 (NULL 허용)
    linked_recruitment_id INT NULL COMMENT '연결된 모집글 ID (모집글 삭제 시 방 삭제용)',
    user1_id INT NULL COMMENT '친구1 ID (친구 탈퇴 시 방 삭제용)',
    user2_id INT NULL COMMENT '친구2 ID (친구 탈퇴 시 방 삭제용)',

    PRIMARY KEY (room_id),

    -- [제약조건] 각 테이블과 연결 (CASCADE 설정)
    CONSTRAINT fk_chat_recruit FOREIGN KEY (linked_recruitment_id) REFERENCES recruitments(id) ON DELETE CASCADE,
    CONSTRAINT fk_chat_user1 FOREIGN KEY (user1_id) REFERENCES users(id) ON DELETE CASCADE,
    CONSTRAINT fk_chat_user2 FOREIGN KEY (user2_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 2. 채팅 메시지 테이블 (수정됨)

CREATE TABLE chat_message (
    message_id BIGINT AUTO_INCREMENT NOT NULL COMMENT '메시지 고유 ID', 
    room_id VARCHAR(50) NOT NULL COMMENT '채팅방 ID (FK: chat_room.room_id)', -- INT에서 VARCHAR로 변경됨
    sender VARCHAR(50) NOT NULL COMMENT '보낸 사람',
    message TEXT COMMENT '메시지 내용',
    message_type VARCHAR(20) NOT NULL COMMENT '메시지 타입 (ENTER, TALK)',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '전송 시간',
    
    PRIMARY KEY (message_id),
    
    -- chat_room의 room_id가 VARCHAR로 바뀌었으므로 여기도 타입이 맞아야 함
    FOREIGN KEY (room_id) REFERENCES chat_room (room_id) ON DELETE CASCADE
);

-- 방문 히스토리 테이블
create table visit_history(
	freeboards int default 0,
    galleries int default 0,
    created_at datetime default now()
);

-- 방문 히스토리 테이블에 트리거 적용 freeboards
DELIMITER $$
CREATE TRIGGER trigger_visit_history_freeboards
AFTER UPDATE ON freeboards
FOR EACH ROW
BEGIN
    IF NEW.view_count > OLD.view_count THEN
        INSERT INTO visit_history (freeboards, galleries, created_at)
        VALUES (1, 0, NOW());
    END IF;
END$$
DELIMITER ;

-- 방문 히스토리 테이블에 트리거 적용 galleries
DELIMITER $$
CREATE TRIGGER trigger_visit_history_galleries
AFTER UPDATE ON galleries
FOR EACH ROW
BEGIN
    IF NEW.view_count > OLD.view_count THEN
        INSERT INTO visit_history (freeboards, galleries, created_at)
        VALUES (0, 1, NOW());
    END IF;
END$$
DELIMITER ;
